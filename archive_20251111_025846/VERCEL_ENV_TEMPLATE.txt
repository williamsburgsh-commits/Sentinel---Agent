# ============================================
# VERCEL ENVIRONMENT VARIABLES - MAINNET
# ============================================
# Copy these to: Vercel Dashboard ‚Üí Settings ‚Üí Environment Variables
# Add each variable one by one using the "Add" button

# ============================================
# 1. NETWORK (REQUIRED)
# ============================================
NEXT_PUBLIC_NETWORK=mainnet

# ============================================
# 2. SOLANA RPC ENDPOINTS (REQUIRED)
# ============================================
NEXT_PUBLIC_MAINNET_RPC=https://api.mainnet-beta.solana.com
NEXT_PUBLIC_DEVNET_RPC=https://api.devnet.solana.com

# ============================================
# 3. SUPABASE (REQUIRED - YOU FILL THESE IN)
# ============================================
# Get from: Supabase Dashboard ‚Üí Settings ‚Üí API

# Project URL (MUST include https://)
NEXT_PUBLIC_SUPABASE_URL=https://YOUR-PROJECT.supabase.co

# Anon/Public Key (starts with eyJ...)
NEXT_PUBLIC_SUPABASE_ANON_KEY=eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...YOUR_KEY_HERE

# ============================================
# HOW TO ADD IN VERCEL
# ============================================
# 1. Go to: https://vercel.com/dashboard
# 2. Select your Sentinel Agent project
# 3. Click: Settings ‚Üí Environment Variables
# 4. For EACH variable above:
#    - Click "Add" button
#    - Name: Copy variable name (e.g., NEXT_PUBLIC_NETWORK)
#    - Value: Copy value (e.g., mainnet)
#    - Environment: Select "Production, Preview, and Development"
#    - Click "Save"
# 5. After adding all variables, go to Deployments tab
# 6. Click "..." on latest deployment ‚Üí "Redeploy"
